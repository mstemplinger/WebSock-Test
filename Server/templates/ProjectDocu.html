def get_column_lengths(table_name):

    try:

        logging.info(f"‚úÖ Tabelle `{table_name}` gefunden. Starte Spaltenpr√ºfung...")

        logging.debug(f"üìä Erkannte Feldl√§ngen f√ºr `{table_name}`: {column_lengths}")
        return column_lengths

    except Exception as e:
        logging.error(f"üö® Fehler in `get_column_lengths({table_name})`: {str(e)}")

def truncate_values(column_values, column_lengths):
    """K√ºrzt die Werte basierend auf den maximalen Feldl√§ngen."""
    logging.info(f"üîç truncate_values `{column_values}`")
    for column, max_length in column_lengths.items():
        if column in column_values and isinstance(column_values[column], str):
            original_value = column_values[column]
            truncated_value = original_value[:max_length]
            column_values[column] = truncated_value
            if original_value != truncated_value:
                logging.warning(f"‚ö†Ô∏è Wert f√ºr `{column}` wurde von {len(original_value)} auf {max_length} Zeichen gek√ºrzt!")
    return column_values

    """Liefert eine Liste der verf√ºgbaren Skripte im Skriptverzeichnis (inkl. Unterverzeichnisse 1. Ebene)"""
    allowed_extensions = {".ps1", ".bat", ".py", ".sh", ".txt"}
    ignored_dirs = {"_obsolete_"}

    if not os.path.exists(SCRIPT_DIR):
        return jsonify({"error": "Skriptverzeichnis nicht gefunden!"}), 500

    try:
        scripts = []

        # üîç Durchlaufe das Hauptverzeichnis
        for root, dirs, files in os.walk(SCRIPT_DIR):
            # Nur die erste Ebene der Unterverzeichnisse betrachten
            rel_root = os.path.relpath(root, SCRIPT_DIR)  # Relativer Pfad zum Root
            if rel_root in ignored_dirs:
                continue  # üö´ Verzeichnis _obsolete_ √ºberspringen

            # üîÑ Unterverzeichnisse nach erster Ebene abschneiden
            if os.path.dirname(rel_root):
                continue  # Verzeichnisse aus tieferen Ebenen ignorieren

            for file in files:
                ext = os.path.splitext(file)[1]
                if ext in allowed_extensions:
                    script_path = os.path.join(rel_root, file) if rel_root != "." else file  # üéØ Pfad zusammensetzen
                    scripts.append({
                        "name": script_path.replace("\\", "/"),  # Konsistente Pfade f√ºr Windows/Linux
                        "type": "powershell" if ext == ".ps1" else
                                "bat" if ext == ".bat" else
                                "python" if ext == ".py" else
                                "linuxshell" if ext == ".sh" else
                                "text"
                    })

        return jsonify(scripts)

    except Exception as e:
        return jsonify({"error": f"Fehler beim Abrufen der Skripte: {str(e)}"}), 500

async def handle_client(websocket):
    """Verarbeitet eingehende WebSocket-Verbindungen von Clients mit erweitertem Logging"""
    try:
        client_ip = websocket.remote_address[0] if websocket.remote_address else "Unbekannt"
        logging.info(f"üîå Neuer Client verbunden von {client_ip}")

        async for message in websocket:
            try:
                logging.info(f"üì© Eingehende Nachricht von {client_ip}: {message}")

                data = json.loads(message)
                action = data.get("action")

                if action == "register":
                    client_id = data.get("client_id")
                    hostname = data.get("hostname")
                    ip_address = data.get("ip")

                    # **Datenvalidierung**
                    if not client_id or not hostname or not ip_address:
                        logging.warning(f"‚ö†Ô∏è Ung√ºltige Registrierungsdaten von {client_ip}: {data}")
                        await websocket.send(json.dumps({"status": "error", "message": "Invalid registration data"}))
                        continue

                    # **Client im lokalen Dictionary speichern**
                    clients[client_id] = {"websocket": websocket, "hostname": hostname, "ip": ip_address}
                    logging.info(f"üì• Neuer Client zwischengespeichert: {clients[client_id]}")

                    # **‚úÖ Datenbank-Operationen f√ºr `acx_asset`**
                    with app.app_context():
                        try:
                            logging.info(f"üîé Suche nach bestehendem Asset f√ºr Client {client_id}...")
                            existing_asset = Asset.query.filter_by(client_id=client_id).first()

                            if existing_asset:
                                logging.info(f"üîÑ Update bestehendes Asset: {existing_asset.client_id} (Last Seen: {existing_asset.last_seen})")
                                existing_asset.last_seen = datetime.now()
                            else:
                                logging.info(f"üÜï Neues Asset wird erstellt f√ºr Client {client_id} ({hostname}, {ip_address})")
                                new_asset = Asset(client_id=client_id, hostname=hostname, ip_address=ip_address)
                                db.session.add(new_asset)

                            db.session.commit()
                            check_for_refresh()  # üîÑ √úberpr√ºfe Client-√Ñnderungen f√ºr Refresh
                            logging.info(f"‚úÖ Client {client_id} erfolgreich registriert oder aktualisiert in `acx_asset`")

                        except SQLAlchemyError as e:
                            db.session.rollback()
                            logging.error(f"‚ùå Datenbankfehler bei Registrierung von {client_id}: {str(e)}")
                            await websocket.send(json.dumps({"status": "error", "message": f"Database error: {str(e)}"}))
                            continue
                        except Exception as e:
                            db.session.rollback()
                            logging.error(f"‚ùå Unerwarteter Fehler bei DB-Operation f√ºr {client_id}: {str(e)}")
                            await websocket.send(json.dumps({"status": "error", "message": f"Unexpected error: {str(e)}"}))
                            continue

                    # **‚úÖ Best√§tigung an den Client senden**
                    response = json.dumps({"status": "registered"})
                    await websocket.send(response)
                    check_for_refresh()  # üîÑ √úberpr√ºfe Client-√Ñnderungen f√ºr Refresh
                    logging.info(f"üì§ Registrierungsbest√§tigung an {hostname} ({ip_address}) gesendet")

                else:
                    logging.warning(f"‚ö†Ô∏è Unbekannte Aktion von {client_ip}: {data}")
                    check_for_refresh()  # üîÑ √úberpr√ºfe Client-√Ñnderungen f√ºr Refresh
                    await websocket.send(json.dumps({"status": "error", "message": "Unknown action"}))

            except json.JSONDecodeError:
                logging.error(f"üö® Ung√ºltiges JSON von {client_ip}: {message}")
                await websocket.send(json.dumps({"status": "error", "message": "Invalid JSON"}))
            except Exception as e:
                logging.error(f"‚ùå Allgemeiner Fehler in der Nachricht von {client_ip}: {str(e)}")
                await websocket.send(json.dumps({"status": "error", "message": f"Processing error: {str(e)}"}))

    except websockets.exceptions.ConnectionClosed as e:
        logging.info(f"‚ùå WebSocket-Verbindung mit {client_ip} geschlossen: {e}")

    finally:
        # **Client aus der DB entfernen, wenn Verbindung verloren geht**
        with app.app_context():
            disconnected_clients = [key for key, value in clients.items() if value["websocket"] == websocket]

            for client_id in disconnected_clients:
                try:
                    logging.info(f"üö™ Entferne Client {client_id} aus Clientspeicher...")
                    clients.pop(client_id, None)

                    logging.info(f"üîé Suche nach Asset {client_id} in `acx_asset` zur L√∂schung...")
                    asset = Asset.query.filter_by(client_id=client_id).first()

                        # db.session.delete(asset)

                        logging.info(f"‚úÖ Client {client_id} erfolgreich aus `acx_asset` entfernt.")
                    else:
                        logging.warning(f"‚ö†Ô∏è Kein Asset-Eintrag f√ºr {client_id} gefunden. Kein L√∂schvorgang durchgef√ºhrt.")

                except SQLAlchemyError as e:
                    db.session.rollback()
                    logging.error(f"‚ùå Fehler beim L√∂schen von Client {client_id} aus `acx_asset`: {str(e)}")
                except Exception as e:

                switch ($scriptType) {

            # JSON-Parsing mit Fehlerbehandlung
            try {

            } catch {

                $data = $null

                        } else {
                            Write-Log "üì© Nachricht vom Server: $($data.content)"
                            Start-Process -FilePath "msg.exe" -ArgumentList "* $($data.content)"
                        }
                    }
                    "upload_script_chunk" {

                    }
                    default {

                    }
                }
            } else {
                Write-Log "‚ö†Ô∏è Nachricht kein g√ºltiges JSON oder kein 'action'-Feld: $message"
            }

        } catch {

    }
